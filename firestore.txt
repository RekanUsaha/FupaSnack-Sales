rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Users collection - hanya bisa baca/edit data sendiri
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Sales collection - karyawan hanya akses data sendiri, admin akses semua
    match /sales/{saleId} {
      allow create: if request.auth != null && 
        request.resource.data.createdBy == request.auth.uid;

      allow read, update, delete: if request.auth != null && 
        (resource.data.createdBy == request.auth.uid || 
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'Admin');
    }

    // BelumDibayar collection - sama seperti sales
    match /belumDibayar/{docId} {
      allow create: if request.auth != null && 
        request.resource.data.createdBy == request.auth.uid;

      allow read, update, delete: if request.auth != null && 
        (resource.data.createdBy == request.auth.uid || 
         get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'Admin');
    }

    // SupplierMaster collection - bisa diakses semua user untuk baca/tulis
    match /supplierMaster/{docId} {
      allow read, write: if request.auth != null;
    }

    // Collections hanya untuk Admin (jika masih digunakan)
    match /gaji/{docId} {
      allow read, write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'Admin';
    }

    match /pengeluaranLainnya/{docId} {
      allow read, write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'Admin';
    }

    match /kasSaatIni/{docId} {
      allow read, write: if request.auth != null && 
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'Admin';
    }
  }
}
